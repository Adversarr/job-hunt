## 为什么（2-3 个因果链）

### 1. LLM 固有局限 → 无法独立完成实际操作

**现象**：LLM 只能生成文本，无法直接调用 API、执行支付、操作数据库  
**根因**：
- LLM 本质是"文本生成器"（next token prediction），没有执行能力
- 没有权限系统，无法验证用户身份和操作授权
- 没有状态记忆，无法跟踪多步骤任务的进度
- 没有事务机制，无法保证操作原子性和一致性

**结果**：必须依赖外部系统完成"动手"部分

### 2. 复杂任务特性 → 需要任务分解与编排

**现象**：行程规划包含多个相互依赖的子任务（搜索→决策→预订→支付）  
**根因**：
- 任务间存在依赖关系（必须先订票才能支付）
- 子任务可能失败（票已售罄、支付失败）
- 需要跨回合保持上下文（用户中途修改需求）
- 需要处理异常和回滚（订了机票但酒店失败需退款）

**结果**：必须有任务编排系统、状态机和持久化存储

### 3. 生产环境要求 → 需要安全与容错机制

**现象**：线上系统必须保证安全、可靠、可审计  
**根因**：
- 支付操作涉及资金安全，需要权限校验和审计
- 第三方 API 可能失败，需要重试和降级
- 用户可能中途取消，需要清理已完成的操作
- 系统故障需要恢复机制

**结果**：必须有权限系统、事务管理、回滚策略

```python
class TaskPlanner:

        {{
            "tasks": [
```

```python
            "auth_required": tool.auth_required,
            "rate_limit": tool.rate_limit,
            "timeout": tool.timeout
        }

        """执行工具调用"""
        # 1. 权限检查
        if self.tools[tool_name]["auth_required"]:
            if not self.check_permission(context["user"], tool_name):
                raise PermissionError(f"No permission for {tool_name}")
        
        # 2. 参数验证
        validated_params = self.validate_params(tool_name, params)
        
        # 3. 执行（带重试）
        result = await self.retry_policy.execute(
            self.tools[tool_name]["instance"].run,
            validated_params,
            timeout=self.tools[tool_name]["timeout"]
        )
        
        # 4. 审计日志
        self.log_execution(tool_name, params, result, context)
        
        return result
```

```python
class StateMachine:
    """任务状态机"""
    
    def __init__(self, initial_state: str, state_definitions: dict):
        self.current_state = initial_state
        self.states = state_definitions
        self.history = []
        self.context = {}
    
    def transition(self, event: str, payload: dict = None):
        """状态转移"""
        # 1. 检查转移合法性
        if event not in self.states[self.current_state]["transitions"]:
            raise IllegalTransitionError(
                f"Cannot transition from {self.current_state} on {event}"
            )
        
        # 2. 执行退出动作
        if "on_exit" in self.states[self.current_state]:
            self.states[self.current_state]["on_exit"](self.context)
        
        # 3. 更新状态
        old_state = self.current_state
        self.current_state = event
        self.history.append((old_state, event, payload))
        
        # 4. 执行进入动作
        if "on_enter" in self.states[self.current_state]:
            self.states[self.current_state]["on_enter"](self.context, payload)
        
        return self.current_state
    
    def can_rollback(self) -> bool:
        """是否可回滚"""
        return len(self.history) > 0

class ConversationMemory:
    """多轮对话记忆"""
    
    def __init__(self, max_turns: int = 10):
        self.turns = []
        self.max_turns = max_turns
        self.entity_memory = {}  # 实体槽位
    
    def add_turn(self, role: str, content: str, metadata: dict = None):
        """添加一轮对话"""
        self.turns.append({
            "role": role,
            "content": content,
            "timestamp": time.time(),
            "metadata": metadata or {}
        })
        
        # 滑动窗口
        if len(self.turns) > self.max_turns:
            self.turns = self.turns[-self.max_turns:]
    
    def get_context(self, include_entities: bool = True) -> str:
        """获取上下文"""
        context = "\n".join([
            f"{turn['role']}: {turn['content']}"
            for turn in self.turns
        ])
        
        if include_entities and self.entity_memory:
            context = f"[当前状态]\n{json.dumps(self.entity_memory, ensure_ascii=False)}\n\n{context}"
        
        return context
    
    def update_entity(self, key: str, value: Any):
        """更新实体槽位"""
        self.entity_memory[key] = value
```

```python
class AuthManager:
```

#### 步骤 5：失败回滚机制

```python
class TransactionManager:
    """分布式事务管理器"""
    
    def __init__(self):
        self.active_transactions = {}
        self.compensation_actions = {}
    
    async def execute_with_compensation(
        self, 
        operations: List[Operation],
        compensation: List[Operation]
    ) -> TransactionResult:
        """执行带补偿的事务"""
        transaction_id = generate_uuid()
        executed = []
        
        try:
            # 顺序执行操作
            for op in operations:
                result = await op.execute()
                executed.append(op)
                self.compensation_actions[transaction_id] = executed
            
            # 全部成功
            return TransactionResult(success=True, transaction_id=transaction_id)
        
        except Exception as e:
            # 失败回滚（逆序执行补偿）
            for op in reversed(executed):
                try:
                    await op.compensate()
                except Exception as comp_e:
                    # 补偿失败，记录并人工介入
                    self.log_compensation_failure(op, comp_e)
            
            return TransactionResult(success=False, error=str(e))

class RollbackHandler:
    """回滚处理器"""
    
    def __init__(self):
        self.strategies = {
            "payment_failed": self.rollback_payment,
            "booking_failed": self.rollback_booking,
            "user_cancelled": self.rollback_all,
        }
    
    async def rollback_payment(self, context: dict):
        """回滚支付"""
        payment_id = context.get("payment_id")
        if payment_id:
            await payment_api.refund(payment_id)
            await notify_user(context["user_id"], "支付已退款")
    
    async def rollback_booking(self, context: dict):

        await notify_user(context["user_id"], "预订已取消")
```

```python
# Redis 状态存储
import redis

state_store = redis.Redis(
    host='localhost',
    port=6379,
    db=0,
    decode_responses=True
)

class StatePersistence:
    """状态持久化"""
    
    def save_state(self, task_id: str, state: dict, ttl: int = 3600):
        """保存状态（带 TTL）"""
        state_store.setex(
            f"task:{task_id}",
            ttl,
            json.dumps(state)
        )
    
    def load_state(self, task_id: str) -> dict:
        """加载状态"""
        data = state_store.get(f"task:{task_id}")
        return json.loads(data) if data else None
    
    def update_state(self, task_id: str, updates: dict):
        """更新状态（原子操作）"""
        with state_store.pipeline() as pipe:
            while True:
                try:
                    pipe.watch(f"task:{task_id}")
                    current = self.load_state(task_id)
                    current.update(updates)
                    pipe.multi()
                    pipe.set(f"task:{task_id}", json.dumps(current))
                    pipe.execute()
                    break
                except redis.WatchError:
                    # 状态被其他进程修改，重试
                    continue
```

## 权衡分析

| 方案 | 收益 | 代价 | 适用边界 |
|------|------|------|----------|
| **纯 LLM（无工具）** | 实现简单、延迟低、无集成成本 | 无法执行实际操作、易幻觉、无状态记忆 | 仅咨询类场景（推荐、问答、创意生成） |
| **LLM + 工具链（无状态机）** | 能执行操作、扩展性强 | 失败无法恢复、复杂任务易混乱、无事务保证 | 简单单步任务（查天气、搜新闻） |
| **LLM + 工具链 + 状态机** | 支持复杂流程、可回滚、可恢复 | 实现复杂、状态管理开销、调试困难 | 多步业务流程（订票、购物） |
| **LLM + 工具链 + 状态机 + 事务** | 强一致性、可靠、可审计 | 最复杂、性能开销大、需要分布式协调 | 涉及资金/权限的关键业务（支付、转账） |
| **多 Agent 协作** | 专业分工、并行处理 | 通信开销、协调复杂、一致性难保证 | 超复杂任务（多领域协同） |

| 架构选择 | 吞吐 | 延迟 | 可靠性 | 实现复杂度 |
|---------|------|------|--------|-----------|
| 单 Agent | 高 | 低 | 中 | 低 |
| 多 Agent 串行 | 低 | 高 | 中 | 中 |
| 多 Agent 并行 | 中 | 中 | 中 | 高 |
| 带 Workflow 引擎 | 中 | 中 | 高 | 高 |

### 3. Q: 如果订票成功但支付失败，如何回滚？补偿事务怎么做？

**A**: 采用 Saga 模式的补偿事务：

```python
# 事务流程
operations = [
    Operation("book_flight", compensate="cancel_flight"),
    Operation("pay_flight", compensate="refund_flight"),
]

# 执行失败回滚
try:
    booking_id = await book_flight(...)  # 成功
    payment_id = await pay_flight(...)   # 失败
except PaymentError:
    # 自动触发补偿
    await cancel_flight(booking_id)
    await notify_user("支付失败，已取消预订")
```

**补偿原则**：
- 补偿操作需幂等（可多次执行）
- 补偿失败需人工介入（告警 + 日志）
- 记录完整审计轨迹

```python
        except Exception as e:
            await self.rollback_all()
    
    async def rollback_all(self):
        """清理所有已执行操作"""
        for log in reversed(self.execution_log):
            try:
                await self.cleanup_action(log["action"], log["result"])
            except Exception as e:
                self.alert_manual_intervention(log, e)
```

### 5. Q: 如何防止用户恶意操作（如订票但不支付）？

**A**: 多层防护机制：

1. **权限校验**：支付前检查用户信用/余额
2. **预扣资源**：订票时锁定资源（5-10分钟），超时自动释放
3. **频率限制**：单用户每分钟最多 N 次操作
4. **行为分析**：检测异常模式（频繁订票取消）
5. **保证金机制**：重要操作需预授权

```python
# 预扣资源示例
class BookingLock:
    def acquire(self, resource_id, user_id, ttl=300):
        """锁定资源（5分钟）"""
        lock_key = f"lock:{resource_id}"
        if redis.setnx(lock_key, user_id):
            redis.expire(lock_key, ttl)
            return True
        return False
    
    def release(self, resource_id):
        """释放锁"""
        redis.delete(f"lock:{resource_id}")
```

### 6. Q: 如何处理工具调用超时或失败？

**A**: 多级容错策略：

```python
class ToolExecutorWithFallback:
    async def execute(self, tool_name, params):
        try:
            # 1. 主工具
            return await self.execute_with_retry(tool_name, params, max_retries=3)
        except TimeoutError:
            # 2. 降级到备用工具
            fallback_tool = self.get_fallback(tool_name)
            return await fallback_tool.run(params)
        except Exception as e:
            # 3. 缓存兜底
            cached = self.cache.get(tool_name, params)
            if cached:
                return cached
            # 4. 返回友好错误
            return {"error": "服务暂时不可用，请稍后重试"}
    
    async def execute_with_retry(self, tool_name, params, max_retries):
        for i in range(max_retries):
            try:
                return await self.tools[tool_name].run(params)
            except Exception as e:
                if i == max_retries - 1:
                    raise
                await asyncio.sleep(2 ** i)  # 指数退避
```

### 7. Q: 多 Agent 协作时，如何保证任务不重复执行？

**A**: 分布式锁 + 任务队列：

```python
import redis
from celery import Celery

# Celery 任务队列
app = Celery('tasks', broker='redis://localhost:6379/0')

@app.task(bind=True)
def process_task(self, task_id):
    # 分布式锁
    lock = redis.Redis().lock(f"task_lock:{task_id}", timeout=300)
    
    if lock.acquire(blocking=False):
        try:
            # 执行任务
            result = execute_task(task_id)
            return result
        finally:
            lock.release()
    else:
        # 任务已被其他 Agent 执行
        return {"status": "already_processed"}
```

## 常见错误（至少 3 个）

### 错误 1：工具描述不清晰，LLM 频繁选错工具

**错误做法**：
```python
tools = [
    {"name": "search", "description": "搜索"}  # 描述过于模糊
]
```

**正确做法**：
```python
tools = [
    {
        "name": "search_flight",
        "description": "搜索航班信息，返回航班号、价格、时间等",
        "parameters": {
            "origin": "出发地（城市名）",
            "destination": "目的地（城市名）",
            "date": "日期（YYYY-MM-DD）"
        },
        "when_to_use": "当用户询问航班、机票、飞机出行时使用"
    }
]
```

**原因**：LLM 需要清晰的语义信息来匹配工具，模糊描述导致推理错误。

### 错误 2：没有状态持久化，用户刷新页面后任务丢失

**错误做法**：
```python
# 只在内存中维护状态
class Agent:
    def __init__(self):
        self.state = {}  # 进程重启就丢失
```

**正确做法**：
```python
class Agent:
    def __init__(self, task_id):
        self.task_id = task_id
        self.state_store = RedisStateStore()
    
    def get_state(self):
        # 从持久化存储加载
        return self.state_store.load(self.task_id)
    
    def update_state(self, updates):
        state = self.get_state()
        state.update(updates)
        self.state_store.save(self.task_id, state)
```

**原因**：生产环境需要支持跨会话恢复、故障恢复、多实例共享状态。

### 错误 3：没有超时保护，工具调用卡死导致整个任务阻塞

**错误做法**：
```python
result = await tool.run(params)  # 无超时限制
```

**正确做法**：
```python
try:
    result = await asyncio.wait_for(
        tool.run(params),
        timeout=30  # 30 秒超时
    )
except asyncio.TimeoutError:
    # 降级处理
    result = await fallback_tool.run(params)
    # 或返回错误
    result = {"error": "操作超时，请稍后重试"}
```

**原因**：第三方 API 可能响应慢或无响应，必须设置超时保护整个系统。

### 错误 4：回滚逻辑缺失，失败后系统状态不一致

**错误表现**：
- 订了机票但支付失败，机票没取消
- 预扣了余额但任务失败，余额未退回
- 用户取消订单但部分操作无法撤销

**正确做法**：
```python
# 每个操作都注册补偿函数
operations = [
    ("book_flight", cancel_flight),
    ("pay_flight", refund_payment),
    ("book_hotel", cancel_hotel),
]

async def execute_with_compensation(operations):
    executed = []
    try:
        for op, _ in operations:
            result = await op()
            executed.append(result)
    except Exception:
        # 失败回滚
        for op_result, (_, compensate) in zip(executed, operations):
            await compensate(op_result)
```

### 错误 5：权限校验遗漏，导致越权操作

**错误表现**：
- 用户 A 能取消用户 B 的订单
- 未登录用户能执行支付操作
- 普通用户能调用管理员接口

**正确做法**：
```python
@require_permission("booking:write")
async def book_flight(user, params):
    # 权限装饰器自动校验
    # 无权限抛出 PermissionDenied
    pass

# 或手动校验
if not auth_manager.check_permission(user, "booking:write"):
    raise PermissionDenied("无权限执行此操作")
```

```python
def transition(self, event, payload=None):
    if event not in self.states[self.current_state]["transitions"]:
        raise IllegalTransitionError(f"Cannot {event} from {self.current_state}")
    
    old_state = self.current_state
    self.current_state = event
    self.history.append((old_state, event, payload))
    
    if "on_enter" in self.states[event]:
        self.states[event]["on_enter"](self.context, payload)
```
